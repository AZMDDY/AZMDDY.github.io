## 思路

这个题目实际上是一个有向图的问题，即从一个节点是否能够访问图中所有的节点。可以使用`深度优先搜索`和`广度优先搜索`来解决。

## 实现

### C++

#### 深度优先搜索

```cpp
#include <vector>

using namespace std;

class Solution {
public:
    vector<bool> visited;
    int visitedNum;

    bool canVisitAllRooms(const vector<vector<int>>& rooms)
    {
        int roomsNum = rooms.size();
        visited.resize(roomsNum, false);
        visitedNum = 0;
        Dfs(rooms, 0);
        return visitedNum == roomsNum;
    }

    void Dfs(const vector<vector<int>>& rooms, int num)
    {
        visited[num] = true;
        visitedNum++;
        for (auto room : rooms[num]) {
            if (!visited[room]) {
                Dfs(rooms, room);
            }
        }
    }
};
```

#### 广度优先搜索

```cpp
class Solution {
public:
    bool canVisitAllRooms(vector<vector<int>>& rooms)
    {
        vector<bool> visited;
        int visitedNum = 0;
        int roomsNum   = rooms.size();
        visited.resize(roomsNum, false);
        queue<int> que;
        que.push(0);
        visited[0] = true;
        while (!que.empty()) {
            int room = que.front();
            que.pop();
            visitedNum++;
            for (auto iter : rooms[room]) {
                if (!visited[iter]) {
                    que.push(iter);
                    visited[iter] = true;
                }
            }
        }
        return visitedNum == roomsNum;
    }
};
```

### Python

#### 深度优先搜索

```python
class Solution:

    def __init__(self):
        self.visited = set()
        self.visitedNum = 0

    def dfs(self, rooms: List[List[int]], num: int):
        self.visited.add(num)
        self.visitedNum += 1
        for room in rooms[num]:
            if room not in self.visited:
                self.dfs(rooms, room)

    def canVisitAllRooms(self, rooms: List[List[int]]) -> bool:
        roomsNum = len(rooms)
        self.dfs(rooms, 0)
        return self.visitedNum == roomsNum
```

#### 广度优先搜索

```python
import collections

class Solution:

    def canVisitAllRooms(self, rooms: List[List[int]]) -> bool:
        roomsNum = len(rooms)
        visited = {0}
        visitedNum = 0
        que = collections.deque([0])
        while que:
            room = que.popleft()
            visitedNum += 1
            for iter in rooms[room]:
                if iter not in visited:
                    visited.add(iter)
                    que.append(iter)
        return visitedNum == roomsNum
```

func canVisitAllRooms(rooms [][]int) bool {
	roomsNum := len(rooms)
	visitedNum := 0
	visited := make([]bool, roomsNum)
	visited[0] = true
	que := []int{}
	que = append(que, 0)
	for i := 0; i < len(que); i++ {
		num := que[i]
		visitedNum++
		for _, iter := range rooms[num] {
			if !visited[iter] {
				visited[iter] = true
				que = append(que, iter)
			}
		}
	}
	return roomsNum == visitedNum
}
